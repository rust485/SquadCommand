/**
 * Display the current games 
 * @param {*} msgGames 
 */
function displayGames(msgGames)
{
	let div = $("#games");
	div.empty();
	let t = $("<table></table>");
	t.append("<tr> <th>Game Id</th> <th>Player 1</th> <th>Player 2</th> <th></th> </tr>");
	games = {};
	msgGames.forEach((g) =>
	{
		games[g.id] = g;
		let row = $("<tr></tr>",
		{
			id: g.id,
			click: function()
			{
				setSelectedDisplay(this);
			}
		});
		row.append("<td>" + g.id + "</td>");
		row.append("<td>" + g.p1id + "</td>");
		row.append("<td>" + g.p2id + "</td>");
		let c = $("<td></td>");
		c.append($("<button/>",
		{
			type: 'button',
			id: g.id,
			text: 'end',
			click: function()
			{
				socket.emit("end game",
				{
					id: this.id
				});
			}
		}));
		row.append(c);
		t.append(row);
	});

	div.append(t);
}
