/**
 * Changetypes dictate state of change to be made to the game
 */
changeType = {
	PLAYER_REMOVED: 0,
	PLAYER_ADDED: 1,
	GAME_REMOVED: 2,
	GAME_ADDED: 3,
	S_GAME_CHANGE: 4
};

/**
 * Model portion of MVC layout of admin
 */
class Model
{
	/**
	 * Constructor for Model 
	 * Initializes games, players, and observer arrays
	 */
	constructor()
	{
		this.games = [];
		this.players = [];
		this.selectedGame = false;
		this.observers = [];
	}

	/**
	 * Notify all observers of the current change
	 * @param {*} change 
	 */
	notifyObservers(change)
	{
		this.observers.forEach((o) => o.update(this, change))
	}

	/**
	 * Add observers to the current admin
	 * @param {*} o 
	 * 	Specified observer
	 */
	registerObserver(o)
	{
		this.observers.push(o);
		o.update(this);
	}

	/**
	 * Remove specified observer to the admin
	 * @param {*} o 
	 * 	Specified observer
	 */
	unregisterObserver(o)
	{
		this.observers = this.obversvers.filter((observer) => observer != o);
	}

	/**
	 * Remove player from game
	 * @param {Player} p 
	 * 	Specified player
	 */
	removePlayer(p)
	{
		this.players = this.players.filter((player) => player != p);
		this.notifyObservers(changeType.PLAYER_REMOVED);
	}

	/**
	 * Add a player to the game
	 * @param {Player} p 
	 */
	addPlayer(p)
	{
		this.players.push(p);
		this.notifyObservers(changeType.PLAYER_ADDED);
	}

	/**
	 * Add a game 
	 * @param {Game} g 
	 * 	Specified Game
	 */
	addGame(g)
	{
		this.games.push(g);
		this.notifyObservers(changeType.GAME_ADDED);
	}

	/**
	 * Remove the specified game
	 * @param {Game} g
	 * 	Specified game 
	 */
	removeGame(g)
	{
		this.games = this.games.filter((game) => game != g);
		this.notifyObservers(changeType.GAME_REMOVED);
	}

	/**
	 * Set game ID
	 * @param {number} id
	 * 		Set ID for the game
	 */
	setSelectedGame(id)
	{

		for (let i = 0; i < this.games.length; i++)
		{
			if (this.games[i].id === id)
			{
				this.selectedGame = this.games[i];
			}
			return;
		}
		this.notifyObservers(changeType.S_GAME_CHANGE);
	}

	/**
	 * Reset all information for the current game
	 */
	resetSelectedGame()
	{
		this.selectedGame = false;
		this.notifyObservers(changeType.S_GAME_CHANGE);
	}

	/**
	 * Get all information for the selected game
	 */
	getSelectedGame()
	{
		return this.selectedGame;
	}
}
